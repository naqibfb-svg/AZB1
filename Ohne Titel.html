const allowedName = "Sophie";


const results = [

let count = 0;


function login() {

    const name = document.getElementById("nameInput").value.trim();

    const msg = document.getElementById("loginMsg");


    if (name.toLowerCase() === allowedName.toLowerCase()) {


        // SHOW SOPHIE MODE

        document.getElementById("overlay").style.display = "flex";


        setTimeout(() => {

            document.getElementById("overlay").style.display = "none";

        }, 2000);


    } else {

        msg.innerHTML = "❌ Zugriff verweigert. Nur für Sophie erlaubt.";

    }

}


function check() {

    const status = document.getElementById("status");


    let text = results[Math.floor(Math.random() * results.length)];

    count++;


    status.innerHTML = text + "<br><br>Analysen: " + count;


    if (count === 5) {

        setTimeout(() => {

            status.innerHTML = "❤️ Fazit: Sophie ist offiziell unkündbar süß.";

        }, 800);

    }
